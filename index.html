    <title>Judicial Board Case Lookup</title>

    <script>

        function lookupCase() {

            let caseID = document.getElementById("caseID").value;

            if (!caseID) {

                alert("Please enter a Case ID.");

                return;

            }

            fetch(`https://script.google.com/macros/s/AKfycbwIlnhJ6MhgxlbegV4SUWNQI7MyO14ok2UffaMoPAm5y2WNUFei6I7g-1jKQ-Qnkxyo/exec?caseID=${caseID}`)

                .then(response => response.json())

                .then(data => {

                    if (data.error) {

                        document.getElementById("result").innerHTML = `<p style="color: red;">${data.error}</p>`;

                    } else {

                        document.getElementById("result").innerHTML = `

                            <p><strong>Case ID:</strong> ${data.caseID}</p>

                            <p><strong>Respondent:</strong> ${data.respondent}</p>

                            <p><strong>Complainant:</strong> ${data.complainant}</p>

                            <p><strong>Judgment:</strong> ${data.judgment}</p>

                            <p><strong>Penalty:</strong> ${data.penalty}</p>

                            <p><strong>Notes:</strong> ${data.notes}</p>

                            <p><strong>Issued By:</strong> ${data.issuedBy}</p>

                        `;

                    }

                })

                .catch(error => console.error("Error:", error));

        }

    </script>

</head>

<body>

    <h2>Judicial Board Case Lookup</h2>

    <input type="text" id="caseID" placeholder="Enter Case ID">

    <button onclick="lookupCase()">Search</button>

    <div id="result"></div>

</body>
